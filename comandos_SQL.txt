CREATE TABLE Usuario (
  matricula INT PRIMARY KEY,
  isAdmin BOOLEAN,
  nome VARCHAR(255),
  email VARCHAR(255),
  curso VARCHAR(255),
  senha VARCHAR(255)
);

CREATE TABLE Professores (
  id_professor INT PRIMARY KEY,
  nome VARCHAR(255),
  fk_cod_depto INT,
  FOREIGN KEY (fk_cod_depto) REFERENCES Departamentos(cod_depto)
);

CREATE TABLE Disciplinas (
  cod_disciplina INT PRIMARY KEY,
  nome VARCHAR(255),
  fk_cod_depto INT,
  FOREIGN KEY (fk_cod_depto) REFERENCES Departamentos(cod_depto)
);

CREATE TABLE Turmas (
  turma VARCHAR(255),
  periodo VARCHAR(255),
  professor VARCHAR(255),
  horario VARCHAR(255),
  vagas_ocupadas INT,
  total_vagas INT,
  local VARCHAR(255),
  fk_cod_disciplina INT,
  cod_depto INT,
  FOREIGN KEY (fk_cod_disciplina) REFERENCES Disciplinas(cod_disciplina),
  FOREIGN KEY (cod_depto) REFERENCES Departamentos(cod_depto)
);

CREATE TABLE Departamentos (
  cod_depto INT PRIMARY KEY,
  nome VARCHAR(255)
);

CREATE TABLE Avaliacoes (
  codigo_avaliacao INT PRIMARY KEY,
  fk_matricula INT,
  FOREIGN KEY (fk_matricula) REFERENCES Usuario(matricula)
);

CREATE TABLE Denuncias (
  codigo_denuncia INT PRIMARY KEY,
  fk_codigo_avaliacao INT,
  fk_matricula INT,
  FOREIGN KEY (fk_codigo_avaliacao) REFERENCES Avaliacoes(codigo_avaliacao),
  FOREIGN KEY (fk_matricula) REFERENCES Usuario(matricula)
);
